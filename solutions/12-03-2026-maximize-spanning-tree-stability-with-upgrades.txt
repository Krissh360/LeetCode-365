Problem: Maximum spanning tree stability with upgrades

Approach:
We are given a graph with n nodes and a list of edges. Each edge has a stability value s and a flag m indicating whether the edge is mandatory (m = 1) or optional (m = 0). We are also allowed to upgrade up to k optional edges to increase their effective stability.

The goal is to build a connected network while maximizing the minimum stability among the chosen edges.
To solve this, we combine Binary Search with the Disjoint Set Union (DSU) data structure.

First, DSU is used to efficiently track connected components and detect cycles while building the graph. It supports two key operations:

find(x) to locate the representative of a component.
union(x, y) to merge two components.

Before starting the main computation, we check if the mandatory edges themselves form a cycle. If they do, it is impossible to construct a valid spanning structure, so we return -1.

Next, we apply binary search on the answer, where the answer represents the maximum possible minimum stability value. For each candidate stability value mid, we check if it is possible to construct a connected graph that satisfies this requirement.

During the check:

If an edge is mandatory (m = 1), it must be included. If its stability is less than mid, the configuration is invalid.
For optional edges (m = 0), if the stability is already at least mid, we can include them normally.
If the stability is less than mid but doubling it (through an upgrade) can reach mid, we store these edges as upgrade candidates.
After processing the edges, we try to use upgrade candidates to connect remaining components, consuming one upgrade from k each time. If we run out of upgrades or cannot connect all components, the configuration is invalid.

Finally, we verify that all nodes belong to a single connected component. If they do, the candidate stability is feasible.

Binary search then continues to try higher stability values until the maximum feasible value is found.

Time Complexity: O(E log M)
Space Complexity: O(n)


```cpp

class DSU {
public:

    vector<int> parent, rank;

    DSU(int n)
    {
        parent.resize(n);
        rank.resize(n, 1);

        for(int i = 0; i < n; i++)
            parent[i] = i;

    }

    int find (int x)
    {
        if (x == parent[x]) 
            return x;

        return parent[x] = find(parent[x]);
    }

    bool Union(int x, int y)
    {
        int x_parent = find(x);
        int y_parent = find(y);

        if (x_parent == y_parent) 
            return false;

        if(rank[x_parent] > rank[y_parent])
        {
            parent[y_parent] = x_parent;
        } 
        else if(rank[x_parent] < rank[y_parent]) 
        {
            parent[x_parent] = y_parent;
        }
        else 
        {
            parent[x_parent] = y_parent;
            rank[y_parent]++;
        }

        return true;
    }
};

class Solution {
public:

    bool check(int n, vector<vector<int>> & edges, int k, int mid)
    {
        DSU dsu(n);
        vector<vector<int>> upgradeCandidate;
        // upgradeCandidate stores the edges with m = 0 that satisfy the condition to be added in the MST but need to be checked for any cycle formation (redundancy)

        for (auto &edge : edges)
        {
            int u = edge[0], v = edge[1];
            int s = edge[2], m = edge[3];

            if (m == 1)
            {
                if (s < mid) return false;
                dsu.Union(u, v);
            } 

            else
            {
                // already satisifes the condition. No need to upgrade
                if (s >= mid)
                {
                    dsu.Union(u, v);
                }
                
                else if (2*s >= mid)
                {
                    upgradeCandidate.push_back({u, v});
                }
            }
        }

        for (auto &edge : upgradeCandidate)
        {
            int u = edge[0], v = edge[1];

            if (dsu.find(u) != dsu.find(v))
            {
                if (k <= 0) return false;

                dsu.Union(u, v);
                k--;
            }
        }

        int root = dsu.find(0);

        for(int node = 1; node <= n-1; node++) 
        {
            if(dsu.find(node) != root) return false;
        }

        return true;
    }

    int maxStability(int n, vector<vector<int>>& edges, int k) {

        DSU dsu(n);
        
        for (auto edge : edges)
        {
            int u = edge[0], v = edge[1];
            int s = edge[2], m = edge[3];

            // if it is mendatory to add 2 edges but they belong to the same component
            if (m == 1)
            {
                if (dsu.find(u) == dsu.find(v)) return -1;
                dsu.Union(u, v);
            }
        }

        int res = -1;
        int l = 1, r = 2*1e5;

        // simple binary search on answers template
        while (l <= r)
        {
            int mid = l + (r-l) / 2;

            if (check(n, edges, k, mid))
            {
                res = mid;
                l = mid + 1;
            }

            else
            {
                r = mid - 1;
            }
        }

        return res;
    }
};
