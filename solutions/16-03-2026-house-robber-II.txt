Problem: House Robber II

Approach:
In this problem, the houses are arranged in a circle, so the first and last houses are adjacent. That means we cannot rob both together.

So we break the problem into two normal House Robber cases:
rob houses from index 0 to n-2
rob houses from index 1 to n-1

This way, one case excludes the last house and the other excludes the first house. Then we take the maximum of the two answers.

For each case, we use the same DP idea as House Robber:
dp[i] = max(current house + dp[i-2], dp[i-1])

This means:
take the current house and add the best answer from two houses before or skip the current house and keep the previous best answer

We compute this separately for both arrays and return the larger result.

Time Complexity: O(n)
Space Complexity: O(n)


```cpp

class Solution {
public:
    int rob(vector<int>& nums) {

        int n = nums.size();

        if (n == 0) return 0;
        if (n == 1) return nums[0];

        vector<int> v1(nums.begin(), nums.end() - 1);
        vector<int> v2(nums.begin() + 1, nums.end());

        int n1 = v1.size();
        vector<int> dp1(n1, 0);
        dp1[0] = v1[0];

        if (n1 > 1) dp1[1] = max(v1[0], v1[1]);

        for (int i = 2; i < n1; i++)
        {
            int take = v1[i] + dp1[i - 2];
            int drop = dp1[i - 1];
            dp1[i] = max(take, drop);
        }

        int n2 = v2.size();
        vector<int> dp2(n2, 0);
        dp2[0] = v2[0];

        if (n2 > 1) dp2[1] = max(v2[0], v2[1]);

        for (int i = 2; i < n2; i++)
        {
            int take = v2[i] + dp2[i - 2];
            int drop = dp2[i - 1];
            dp2[i] = max(take, drop);
        }

        return max(dp1[n1 - 1], dp2[n2 - 1]);
        
    }
};
