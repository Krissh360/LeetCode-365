Problem: 3Sum

Approach:
Sorted the array to enable the two-pointer technique.
Fixed one element at a time and searched for the remaining two elements using two pointers.
For each fixed element nums[i], set a target as -nums[i].
Initialized two pointers: left at i+1 and right at the end of the array.
Moved pointers inward based on the sum comparison with the target.
Skipped duplicate elements to ensure unique triplets.

Time Complexity: O(n^2)
Space Complexity: O(1)

```cpp

class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        
        vector<vector<int>> ans;
        int n = nums.size();

        sort(nums.begin(), nums.end());

        for (int i = 0; i < n; i++)
        {

            if (i > 0 && nums[i] == nums[i - 1]) continue;

            int target = 0 - nums[i];
            int j = i+1, k = n-1;

            while (j < k)
            {
                if (i != j && j != k && k!= i)
                {
                    if (nums[j] + nums[k] == target)
                    {
                       ans.push_back({nums[i], nums[j], nums[k]});
                       j++;
                       k--;

                       while (j < k && nums[j] == nums[j - 1]) j++;
                       while (j < k && nums[k] == nums[k + 1]) k--;
                    }

                    else if(nums[j] + nums[k] < target) {j++;}
                    else {k--;}
                }
            }
        }

        return ans;
    }
};
