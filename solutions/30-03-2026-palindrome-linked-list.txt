Problem: Palindrome Linked List

Approach:
We need to check if a linked list reads the same forward and backward.

First, we find the middle of the list using slow and fast pointers.
Then we reverse the second half of the list.

Now we compare the first half and the reversed second half node by node.
- If any values differ, it is not a palindrome

After comparison, we restore the original list by reversing the second half again.

If all values match, the list is a palindrome.

Time Complexity: O(n)
Space Complexity: O(1)


```cpp


class Solution {
public:

    ListNode* revLL (ListNode* head)
    {
        if (!head || !head -> next) return head;

        ListNode* newHead = revLL(head -> next);
        ListNode* front = head -> next;
        front -> next = head;
        head -> next = NULL;

        return newHead;
    }

    bool isPalindrome(ListNode* head) {

        if (!head || !head -> next) return head;
        
        ListNode* slow = head;
        ListNode* fast = head;

        while (fast -> next != NULL && fast -> next -> next != NULL)
        {
            slow = slow -> next;
            fast = fast -> next -> next;
        }

        ListNode* newHead = revLL(slow -> next);
        ListNode* first = head;
        ListNode* second = newHead;

        while (second != NULL)
        {
            if (first -> val != second -> val)
            {
                revLL(newHead);
                return false;
            }

            first = first -> next;
            second = second -> next;
        }

        revLL(newHead);
        return true;
        
    }
};
