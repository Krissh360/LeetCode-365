Problem: Generate String Based on Conditions

Approach:
We are given two strings:
- s1 tells whether a substring should match (T) or not match (F)
- s2 is the pattern string

We need to build a string such that:
- For every 'T' in s1, the substring matches s2
- For every 'F', the substring does not match s2

First, we create a result string filled with '#'.

For all 'T' cases:
- We force the substring to match s2
- If there is a conflict while placing characters, we return ""

Then we fill remaining '#' with 'a' (default choice) and mark them as changeable.

For all 'F' cases:
- If the substring matches s2, we try to break it
- We change one character (from right to left) at a position that we are allowed to modify
- If we cannot break the match, return ""

Finally, return the constructed string.

Time Complexity: O(n × m)
Space Complexity: O(n + m)

```cpp

class Solution {
public:

    bool isSame(string &word, string &s2, int i, int m)
    {
        for (int j = 0; j < m; j++)
        {
            if (word[i] != s2[j]) return false;
            i++;
        }

        return true;
    }

    string generateString(string s1, string s2) {

        int n = s1.length();
        int m = s2.length();
        int N = n+m-1;

        vector<bool> canChange(N, false);
        string word(N, '#');

        for (int i = 0; i < n; i++)
        {
            if (s1[i] == 'T')
            {
                int k = i;
                for (int j = 0; j < m; j++)
                {
                    if (word[k] != '#' && word[k] != s2[j])
                        return "";

                    word[k] = s2[j];
                    k++;                   
                }
            }
        }

        for (int i = 0; i < N; i++)
        {
            if (word[i] == '#')
            {
                word[i] = 'a';
                canChange[i] = true;
            }
        }

        for (int i = 0; i < n; i++)
        {
            if (s1[i] == 'F')
            {
                if (isSame(word, s2, i, m))
                {
                    bool changed = false;

                    for (int j = i + m - 1; j >= i; j--)
                    {
                        if (canChange[j])
                        {
                            char original = word[j];
                            word[j] = (word[j] == 'a' ? 'b' : 'a');

                            if (!isSame(word, s2, i, m))
                            {
                                changed = true;
                                break;
                            }

                            word[j] = original;
                        }
                    }

                    if (!changed) return "";
                }
            }
        }

        return word;
        
    }
};
