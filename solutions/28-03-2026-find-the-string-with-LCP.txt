Problem: Find the String With LCP

Approach:
We are given an LCP (Longest Common Prefix) matrix and need to construct a valid string that matches it.

First, we assign groups to indices:
- If lcp[i][j] > 0, it means positions i and j must have the same character
- We group such indices together and assign them the same character
- We use at most 26 groups (characters 'a' to 'z'), otherwise return empty

Next, we build the string:
- Each index gets a character based on its group

Then we validate the string:
- We recompute the LCP matrix using the constructed string
- If it matches the given LCP matrix, the string is valid
- Otherwise, return empty

Time Complexity: O(n²)
Space Complexity: O(n²)

```cpp

class Solution {
public:
    string findTheString(vector<vector<int>>& lcp) {
        int n = lcp.size();
        vector<int> group(n, -1);
        int gid = 0;

        for (int i = 0; i < n; i++) 
        {
            if (group[i] != -1) continue;

            if (gid >= 26) return "";

            for (int j = i; j < n; j++)
            {
                if (lcp[i][j] > 0)
                    group[j] = gid;
            }
            gid++;
        }

        string s(n, 'a');
        for (int i = 0; i < n; i++)
        {
            s[i] = 'a' + group[i];
        }
        
        vector<vector<int>> calc(n, vector<int>(n, 0));

        for (int i = n - 1; i >= 0; i--)
        {
            for (int j = n - 1; j >= 0; j--)
            {
                if (s[i] == s[j])
                {
                    if (i == n - 1 || j == n - 1)
                        calc[i][j] = 1;
                    else
                        calc[i][j] = 1 + calc[i + 1][j + 1];
                }
            }
        }

        if (calc != lcp) return "";

        return s;
    }
};
