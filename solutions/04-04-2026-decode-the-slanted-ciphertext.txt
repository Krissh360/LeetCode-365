Problem: Decode the Slanted Ciphertext

Approach:
We are given an encoded string and the number of rows. The string was originally written row-wise into a matrix and then read diagonally.

First, we calculate the number of columns:
cols = total length / rows

Then we fill a matrix row-wise using the encoded string.

After that, we traverse the matrix diagonally:
- Start from each column in the first row
- Move diagonally down-right and collect characters

This gives us the decoded string.

Finally, we remove any trailing spaces from the result.

Time Complexity: O(n)
Space Complexity: O(n)


```cpp


class Solution {
public:
    string decodeCiphertext(string encodedText, int rows) {
        if (rows == 0) return "";

        int n = encodedText.size();
        int cols = n / rows;

        vector<vector<char>> mat(rows, vector<char>(cols));
        int idx = 0;
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                mat[i][j] = encodedText[idx++];
            }
        }

        string result = "";
        for (int startCol = 0; startCol < cols; startCol++) {
            int i = 0, j = startCol;
            while (i < rows && j < cols) {
                result += mat[i][j];
                i++;
                j++;
            }
        }

        while (!result.empty() && result.back() == ' ') {
            result.pop_back();
        }

        return result;
    }
};
